select  *   from 	   (    ( SELECT   Country,Product,URI,UPC,EAN,ISRC,Track_name,Artist_name,Composer_name,Album_name,Quantity,Label ,  sum(Payable_USD) as Total_Revenue  , 30 as rev_share, 0 as gst_percentage, 0 as holding_percentage 
                FROM report_audio_spotify_2022_01 
                where  report_audio_spotify_2022_01.content_owner = 'mobilemedia'   group by ISRC )   )  as list_all order by Total_Revenue desc LIMIT 0, 2
