select  *   from 	  (  ( SELECT    videoID,channelID,video_title,assetChannelID,assetID,contentType,content_owner,Label, sum(partnerRevenue) as partnerRevenue ,'nd1'  as tablename , 30 as rev_share, 0 as gst_percentage,  holding_percentage FROM youtube_video_claim_report_nd1_2022_12 where  content_owner = 'airnet'   group by videoID )   union    ( SELECT    videoID,channelID,video_title,assetChannelID,assetID,contentType,content_owner,Label, sum(partnerRevenue) as partnerRevenue ,'nd2'  as tablename , 85 as rev_share, 0.00 as gst_percentage,  holding_percentage FROM youtube_video_claim_report_nd2_2022_12 where  content_owner = 'airnet'   group by videoID )   union    ( SELECT    videoID,channelID,video_title,assetChannelID,assetID,contentType,content_owner,Label, sum(partnerRevenue) as partnerRevenue ,'ndkids'  as tablename , 30 as rev_share, 0 as gst_percentage,  holding_percentage FROM youtube_video_claim_report_ndkids_2022_12 where  content_owner = 'airnet'   group by videoID )   )  as list_all order by partnerRevenue desc, tablename desc LIMIT 200, 100
