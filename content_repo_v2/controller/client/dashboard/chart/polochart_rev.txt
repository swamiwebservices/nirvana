select  *   from 	  (  ( SELECT DATE(day) as day,SUM(partnerRevenue) as partnerRevenue,SUM(if(contentType='UGC',partnerRevenue,0)) as UGC,SUM(if(contentType='Partner-provided',partnerRevenue,0)) as Partnerprovided FROM youtuberedmusic_video_report_redmusic_2022_09 where content_owner = 'marsorbit'   group by day ORDER by day  )   )  as list_all order by day asc LIMIT 0, 50
